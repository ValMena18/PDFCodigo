const formatCurrency = (value) => {
  const numericValue = Number(value) || 0;
  return new Intl.NumberFormat('en-US', {
    style: 'decimal',
    minimumFractionDigits: 2,
    maximumFractionDigits: 2
  }).format(numericValue);
};

// Función para formatear fechas
const formatDate = (dateString) => {
  if (!dateString) return 'No especificada';
  try {
    const options = { year: 'numeric', month: 'long', day: 'numeric' };
    return new Date(dateString).toLocaleDateString('es-ES', options);
  } catch {
    return dateString;
  }
};

const ReservationPDF = ({ reservation }) => {


  if (!reservation) {
    return (
      
        
          No hay datos de reserva disponibles
        
      
    );
  }

  
  // Cálculo de totales
  const totalHabitaciones = reservation.habitaciones?.reduce((sum, item) => sum + (Number(item.total) || 0), 0) || 0;
  const totalComplementos = reservation.complementos?.reduce((sum, item) => sum + (Number(item.total) || 0), 0) || 0;
  const subtotal = (reservation.totales?.subtotal) || (totalHabitaciones + totalComplementos);
  const iva = (reservation.totales?.iva) || (subtotal * 0.13);
  const total = (reservation.totales?.total) || (subtotal + iva);

  return (
    
      

        {/* Encabezado */}
        
        
          
            Resumen de Reserva #{reservation.reserva?.id_reserva || 'N/A'}
            Fecha: {formatDate(new Date().toISOString())}
          
        

        {/* Información del Crucero */}
        
          Detalles del Crucero
          
            
              Crucero: 
              {reservation.reserva?.nombre_crucero || 'No especificado'}
            
            
              Pasajeros: 
              {reservation.reserva?.cantidad_pasajeros || 0}
            
            
              Duración: 
              {reservation.reserva?.duracion_dias || 0} días
            
            
              Pago: 
              {reservation.reserva?.metodo_pago || 'No especificado'}
            
          
        

        {/* Itinerario compacto */}
        
          Itinerario
          
            
              Salida:
              {reservation.puertos?.salida || 'No especificado'}
              {formatDate(reservation.reserva?.fecha_inicio)}
            
            
              Regreso:
              {reservation.puertos?.regreso || 'No especificado'}
              {formatDate(reservation.fecha_fin)}
            
          
        

        {/* Habitaciones - Tabla compacta */}
        
          Habitaciones ({reservation.habitaciones?.length || 0})
          
            
              Tipo
              Cant.
              Pers.
              P. Unit.
              Total
            
            
            {reservation.habitaciones.map((habitacion, index) => (
              
                {habitacion.nombre || 'N/A'}
                {habitacion.cantidad || 0}
                {habitacion.cantidad_huespedes || 0}
                $ {formatCurrency(habitacion.precio)}
                $ {formatCurrency(habitacion.total)}
              
            ))}

          
          Total Habitaciones
              
              $ {formatCurrency(totalHabitaciones)}
          
          
        

        {/* Complementos - Solo si existen */}
        {reservation.complementos.length > 0 && (
          
            Servicios Adicionales ({reservation.complementos.length})
            
              
                Servicio
                Cant.
                P. Unit.
                Total
              
              
              {reservation.complementos.map((complemento, index) => (
                
                  {complemento.nombre || 'N/A'}
                  {complemento.cantidad || 0}
                  $ {formatCurrency(complemento.precio)}
                  $ {formatCurrency(complemento.total)}
                
              ))}
              
          
          Total Complementos
              
              $ {formatCurrency(totalComplementos)}
          
            
            
          
        )}

        {/* Totales compactos */}
        
          Resumen de Pagos
          
            
              Subtotal:
              $ {formatCurrency(subtotal)}
            
            
              Impuestos (13%):
              $ {formatCurrency(iva)}
            
            
              TOTAL:
              $ {formatCurrency(total)}
            
          
        

        {/* Estado del Pago compacto */}
        
          ESTADO:
          
          {reservation.reserva?.estado_pago === 'Completado' ? (
            PAGO COMPLETO
          ) : (
            <>
              PENDIENTE DE PAGO
              {reservation.reserva?.fecha_limite_pago && (
                
                  Vence: 
                  {formatDate(reservation.reserva.fecha_limite_pago)}
                
              )}
              
                * Realice el pago antes de la fecha límite
              
            
          )}
        

        {/* Pie de página */}
        
          BALTICO DORADO | Tel: +506 2222-2222 | Email: info@balticodorado.com
        

        
      
    
  );
};

      nombre_crucero: PropTypes.string,

      fecha_inicio: PropTypes.string,
      fecha_limite_pago: PropTypes.string,

    fecha_fin: PropTypes.string,
    puertos: PropTypes.shape({

    habitaciones: PropTypes.arrayOf(

    totales: PropTypes.shape({

};

Descripción del Componente

Este componente React utiliza @react-pdf/renderer para generar documentos PDF con el resumen de reservas de cruceros.

Características Principales

  • Generación de PDFs dinámicos con datos de reserva
  • Formateo de moneda y fechas
  • Diseño responsivo para PDF
  • Tablas para mostrar habitaciones y servicios
  • Cálculo automático de totales e impuestos

Dependencias

  • React
  • @react-pdf/renderer
  • prop-types

Uso

import ReservationPDF from './ReservationPDF';

import { PDFDownloadLink } from '@react-pdf/renderer';
import { Button } from '@mui/material';
import ReservationPDF from './ReservationPDF';

// Datos de ejemplo para la reserva
const reserva = {
  reserva: {
    id_reserva: 'CRU2023-001',
    nombre_crucero: 'Caribe Paradise',
    duracion_dias: 7,
    // ... otros datos de reserva
  },
  habitaciones: [
    // ... datos de habitaciones
  ],
  complementos: [
    // ... datos de complementos
  ]
};

function DownloadButton() {
  return (
    <PDFDownloadLink
      document={<ReservationPDF reservation={reserva} />}
      fileName={`Reserva-Crucero-${reserva.reserva?.id_reserva || 'N/A'}.pdf`}
    >
      {({ loading }) => (
        <Button
          variant="contained"
          color="primary"
          disabled={loading}
          startIcon={loading ? <CircularProgress size={20} /> : <PictureAsPdfIcon />}
        >
          {loading ? 'Generando PDF...' : 'Descargar Factura'}
        </Button>
      )}
    </PDFDownloadLink>
  );
}

Características del Componente

  • PDFDownloadLink: Componente de react-pdf/renderer que maneja la generación y descarga
  • Estado de carga: Muestra feedback visual mientras se genera el PDF
  • Nombre dinámico: El nombre del archivo incluye el ID de reserva
  • Integración con MUI: Usa componentes de Material-UI para mejor apariencia
